#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
//菱形
int main()
{
    int line =0;
    scanf("%d",&line);
    int i=0;
    //打印上半部分
    for(i=0; i<line; i++)
    {
        int j=0;
        for(j=0; j<line-i-1; j++)
        {
            printf(" ");
        }
        for(j=0; j<2*i+1; j++)
        {
            printf("*");
        }
        printf("\n");
    }
    //打印下半部分
    for(i=0; i<line-1; i++)
    {
        int j=0;
        for(j=0; j<=i;j++)
        {
            printf(" ");
        }
        for(j=0; j<2*(line-i-1)-1; j++)
        {
            printf("*");
        }
        printf("\n");
    }

    return 0;
}
