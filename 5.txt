//指针数组
int main()
{
int arr1[]={1,2,3,4,5};
int arr2[]={3,4,5,6,7};
int arr3[]={5,6,7,8,9};
int* parr[]={ arr1, arr2, arr3};
int i = 0;
for (i = 0 ,i < 3, i++)
{
 int j = 0;
 for(j = 0, j<5,j++)
 {
  printf("%d\n", *(parr[i]+j));
  }
}
return 0;
}
//结果 1 2 3 4 5
//        3 4 5 6 7
//        5 6 7 8 9
